%%%===== Glossary for definitions, acronyms and abbreviations ============

\newglossaryentry{biogas}
				{
				name = biogas,
				description = {gaseous mixture, containing mainly methane and carbon dioxide, produced by anaerobic digestion}
				}

\newglossaryentry{biomass}
				{
				name = biomass,
				description = {organic based substrate that may be used as feedstock for anaerobic digestion}
				}

\newglossaryentry{biomethane}
				{
				name = biomethane,
				description = {purified form of biogas containing mainly CH$_4$, having other gases removed via upgrading and cleaning}
				}

\newglossaryentry{cleaning}
				{
				name = cleaning,
				description = {water and hydrogen sulphide removal biogas processing step}
				}

\newglossaryentry{grid}
				{
				name = grid,
				description = {network of transmission infrastructure used to supply a country with utilities, namely gas and electricity}
				}
				
\newglossaryentry{membraneseparation}
				{
				name = membrane separation,
				description = {an upgrading technology using polymer membranes to separate CO$_2$ and CH$_4$ based on differences in permeability}
				}
				
\newglossaryentry{permeability}
				{
				name = permeability,
				description = {the rate at which a substance can travel through a membrane}
				}

\newglossaryentry{selectivity}
				{
				name = selectivity,
				description = {the ratio of permeabilities for different gas species through a given membrane}
				}
				
\newglossaryentry{upgrading}
				{
				name = upgrading,
				description = {carbon dioxide removal from biogas processing step}
				}

\newglossaryentry{waterscrubbing}
				{
				name = water scrubbing,
				description = {absorption column that uses water as the liquid to remove CO$_2$ from biogas, usually operating between $8-10 \, bar$ and ambient temperature}
				}

\newacronym{ad}{AD}{anaerobic digestion}

\newacronym{chp}{CHP}{combined heat and power}

\newacronym{cepci}{CEPCI}{Chemical Engineering Plant Cost Index}

\newacronym{cfd}{CfD}{contracts for difference}

\newacronym{decc}{DECC}{department of energy and climate change}

\newacronym{dbeis}{DBEIS}{department of business, energy and industrial strategy}

\newacronym{fit}{FiT}{feed-in tariff}

\newacronym{gwp}{GWP}{global warming potential}

\newacronym{ghg}{GHG}{green-house gas}

\newacronym{mtoe}{Mtoe}{mega tonnes oil equivalent}

\newacronym{npv}{NPV}{net present value}

\newacronym{psa}{PSA}{pressure swing adsorption}

\newacronym{rec}{REC}{methane recovery}

\newacronym{rev}{REV}{revenue stream}

\newacronym{rhi}{RHI}{renewable heat incentive}

\newacronym{roc}{ROC}{renewable obligation certificate}

\newacronym{rtfc}{RTFC}{renewable transport fuel certificate}

\newacronym{rtfo}{RTFO}{renewable transport fuel obligation}

\newacronym{sem}{SEM}{scanning electron microscopy}
